class Solution {
public:
    double myPow(double x, long n) {
        if (n == 0) return 1.0;
        if (n < 0) {
            x = 1 / x;
            n = labs(n);
        }
        double res = 1.0;
        while (n >= 1) {
            if (n % 2 == 1) {
                res *= x;
            }
            x *= x;
            n /= 2;
        }
        return res;
    }
};

// T.C. = O(log n)
// S.C. = O(1)
